#We hate England more than you
country_event = {
	id = wacky_events.2
	title = "wacky_events.2.t"
	desc = "wacky_events.2.d"
	picture = ACCUSATION_eventPicture
	fire_only_once = yes
	trigger = {
		exists = ENG
		exists = SCO
		exists = IRE
		OR = {
			tag = SCO
			tag = IRE
		}
		ai = no
	}
	
	mean_time_to_happen = {
		months = 240
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					tag = SCO
				}
				IRE = {
					save_event_target_as = insulted_party
				}
			}
			else_if = {
				limit = {
					tag = IRE
				}
				SCO = {
					save_event_target_as = insulted_party
				}
			}
		}
	}
	
	#How dare you? We hate England more!
	option = {
		name = wacky_events.2.a
		event_target:insulted_party = {
			reverse_add_casus_belli = {	
				target = ROOT
				type = cb_insult
				months = 12
			}
		}
		ROOT = {
			reverse_add_casus_belli = {	
				target = event_target:insulted_party
				type = cb_insult
				months = 12
			}
		}
	}
	
	#We both hate England
	option = {
		name = wacky_events.2.b
		event_target:insulted_party = {
			limit = { is_at_war = no }
			add_opinion = { who = ROOT modifier = opinion_eased_tension }
		}
		ROOT = {
			limit = { is_at_war = no }
			add_opinion = { who = event_target:insulted_party modifier = opinion_eased_tension }
		}
	}
